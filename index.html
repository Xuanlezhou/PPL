const SHEET_NAME = "ClassNotes"; function doGet(e) { const action = e.parameter.action; const callback = e.parameter.callback; if (action === 'save') { const data = JSON.parse(decodeURIComponent(e.parameter.data)); saveData(data); const result = JSON.stringify({ status: "ok" }); const output = callback ? callback + '(' + result + ')' : result; return ContentService.createTextOutput(output) .setMimeType(callback ? ContentService.MimeType.JAVASCRIPT : ContentService.MimeType.JSON); } // Load all notes const data = getNotesData(); const result = JSON.stringify(data); const output = callback ? callback + '(' + result + ')' : result; return ContentService.createTextOutput(output) .setMimeType(callback ? ContentService.MimeType.JAVASCRIPT : ContentService.MimeType.JSON); } function saveData(payload) { const sheet = getSheet(); const rows = sheet.getDataRange().getValues(); const keyMap = {}; rows.forEach((row, i) => { if (row[0]) keyMap[row[0]] = i + 1; }); Object.entries(payload).forEach(([key, value]) => { if (keyMap[key]) { sheet.getRange(keyMap[key], 2).setValue(value); } else { sheet.appendRow([key, value]); } }); } function getNotesData() { const sheet = getSheet(); const data = {}; const rows = sheet.getDataRange().getValues(); rows.forEach(row => { if (row[0]) data[row[0]] = row[1]; }); return data; } function getSheet() { let ss; try { ss = SpreadsheetApp.getActiveSpreadsheet(); } catch(e) {} if (!ss) ss = SpreadsheetApp.create("XuanleZhouSchedule"); let sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) sheet = ss.insertSheet(SHEET_NAME); return sheet; }
